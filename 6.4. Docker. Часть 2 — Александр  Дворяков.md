### Задание 4

**Выполните действия:**

1. Установите pgAdmin с именем контейнера <ваши фамилия и инициалы>-pgadmin.
2. Задайте логин администратора pgAdmin <ваши фамилия и инициалы>@ilove-netology.com и пароль на выбор.
3. Назначьте для данного контейнера статический IP из подсети 172.22.0.0/24.
4. Прокиньте на 80 порт контейнера порт 61231.

В качестве решения приложите:

- текст конфига текущего сервиса;
- скриншот админки pgAdmin.
```
version: '3'
services:
  postgres-server:    # The Postgres Database Service
    image: postgres:latest
    restart: always
    container_name: dvoryakovaa-netology-db
    environment:   # Username, password and database name variables
      POSTGRES_USER: zabbix
      POSTGRES_PASSWORD: dvoryakovaa123
      POSTGRES_DB: dvoryakovaa-db
      PG_DATA: /var/lib/postgresql/data/pgdata #data storage
    networks:
      app-network:
        ipv4_address: '172.22.0.2'

  pgadmin:          # The main pgadmin Software Service
    restart: always
    image: dpage/pgadmin4
    container_name: dvoryakovaa-pgadmin
    environment:
      PGADMIN_DEFAULT_EMAIL: dvoryakovaa@ilove-netology.com
      PGADMIN_DEFAULT_PASSWORD: 1234
    ports:
      - "61231:80"
    depends_on:
      - postgres-server
    networks:
      app-network:
        ipv4_address: 172.22.0.3

#Docker Networks
networks:
  app-network:
    ipam:
      driver: default
      config:
        - subnet: 172.22.0.0/24   
```

### Задание 7

**Выполните действия.**

Настройте линки, чтобы контейнеры запускались только в момент, когда запущены контейнеры, от которых они зависят.

В качестве решения приложите:

- текст конфига **целиком**;
- скриншот команды docker ps;
- скриншот авторизации в админке Zabbix.

```
version: '3'
services:
  postgres-server:    # The Postgres Database Service
    image: postgres:latest
    restart: always
    container_name: dvoryakovaa-netology-db
    environment:   # Username, password and database name variables
      POSTGRES_USER: zabbix
      POSTGRES_PASSWORD: dvoryakovaa123
      POSTGRES_DB: dvoryakovaa-db
      PG_DATA: /var/lib/postgresql/data/pgdata #data storage
    networks:
      app-network:
        ipv4_address: '172.22.0.2'

  pgadmin:          # The main pgadmin Software Service
    restart: always
    image: dpage/pgadmin4
    container_name: dvoryakovaa-pgadmin
    environment:
      PGADMIN_DEFAULT_EMAIL: dvoryakovaa@ilove-netology.com
      PGADMIN_DEFAULT_PASSWORD: 1234
    ports:
      - "61231:80"
    depends_on:
      - postgres-server
    networks:
      app-network:
        ipv4_address: 172.22.0.3


  zabbix-server:     # The main Zabbix Server Software Service
    image: zabbix/zabbix-server-pgsql:ubuntu-latest
    restart: always
    container_name: dvoryakovaa-zabbix-netology
    environment:   # The Postgres database value variable
      POSTGRES_USER: zabbix
      POSTGRES_PASSWORD: dvoryakovaa123
      POSTGRES_DB: dvoryakovaa-db
      ZBX_HISTORYSTORAGETYPES: log,text #Zabbix configuration variables
      ZBX_DEBUGLEVEL: 1
      ZBX_HOUSEKEEPINGFREQUENCY: 1
      ZBX_MAXHOUSEKEEPERDELETE: 5000
    depends_on:
      - postgres-server
    volumes:  # Volumes for scripts and related files you can add
      - /usr/lib/zabbix/alertscripts:/usr/lib/zabbix/alertscripts
    networks:
      app-network:
        ipv4_address: '172.22.0.4'

  zabbix-web:    # The main Zabbix web UI or interface 
    image: zabbix/zabbix-web-apache-pgsql
    #image: zabbix/zabbix-web-nginx-pgsql:ubuntu-latest
    restart: always
    container_name: dvoryakovaa-netology-zabbix-frontend
    environment:  # Postgre database variables
      POSTGRES_USER: zabbix
      POSTGRES_PASSWORD: dvoryakovaa123
      POSTGRES_DB: dvoryakovaa-db
      ZBX_SERVER_HOST: zabbix-server  # Zabbix related and Php variables
      ZBX_POSTMAXSIZE: 64M
      ZBX_MAXEXECUTIONTIME: 500
      PHP_TZ: "Europe/Moscow"
    depends_on:
      - postgres-server
      - zabbix-server
    ports:    # Port where Zabbix UI is available
      - 8080:8080
      - 8443:8443
    networks:
      app-network:
        ipv4_address: '172.22.0.5'


#Docker Networks
networks:
  app-network:
    ipam:
      driver: default
      config:
        - subnet: 172.22.0.0/24   
```
